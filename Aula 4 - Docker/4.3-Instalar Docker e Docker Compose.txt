---
- name: Instalar Docker e Docker Compose na VM
  hosts: cliente1
  become: yes
  tasks:
     - name: Atualizar lista de pacotes
       apt:
         update_cache: yes

     - name: Instalar Docker
       apt:
         name: docker.io
         state: present

     - name: Instalar Docker Compose
       apt:
         name: docker-compose
         state: present
