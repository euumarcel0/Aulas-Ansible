---
- name: Instalar Nginx e Node.js na VM
  hosts: cliente1
  become: yes
  tasks:
     - name: Atualizar lista de pacotes
       apt:
         update_cache: yes

     - name: Instalar Nginx
       apt:
         name: nginx
         state: present

     - name: Instalar Node.js
       apt:
         name: nodejs
         state: present
