---
- name: Instalar GIT e Apache na VM
  hosts: cliente1
  become: yes
  tasks:
    - name: Atualizar lista de pacotes
      apt:
        update_cache: yes

    - name: Instalar GIT
      apt:
        name: git
        state: present

    - name: Instalar Apache
      apt:
        name: apache2
        state: present
